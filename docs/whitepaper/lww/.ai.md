## 발표 자료 재생성 방법

```bash
BASE="docs/whitepaper/lww/presentation"

# 1. PPTX 재생성 (이미지 포함)
python $BASE/scripts/generate_pptx.py \
  --data $BASE/content/slides-data.json \
  --output $BASE/slides/lww-pitch-deck.pptx --with-images

# 2. TTS 재생성 (OPENAI_API_KEY 또는 ELEVENLABS_API_KEY 필요)
python $BASE/scripts/generate_tts.py \
  --data $BASE/content/slides-data.json \
  --output-dir $BASE/audio/

# 3. durationInFrames 갱신 + Remotion 데이터 동기화
python $BASE/scripts/update_durations.py \
  --data $BASE/content/slides-data.json \
  --audio-dir $BASE/audio/

# 4. 오디오/이미지 → Remotion public/ 복사
cp $BASE/audio/*.mp3 $BASE/remotion/public/audio/
cp $BASE/images/*.png $BASE/remotion/public/images/

# 5. MP4 렌더링
cd $BASE/remotion
NODE_OPTIONS="--max-old-space-size=8192" npx remotion render LwwPresentation \
  ../output/lww-pitch-video.mp4 --codec h264 --crf 18 --pixel-format yuv420p
```
